33. C++ Program to find volume of cylinder


        #include<iostream>
        using namespace std;
        
        int main() {
            int r,h;
            cout<<"Enter Radius of Cylinder : "<<endl;
            cin>>r;
            cout<<"Enter Height of Cylinder : "<<endl;
            cin>>h;
            float area=3.14*r*r*h;
            cout<<"Volume of Cylinder : "<<area<<" cubic units";
        }
