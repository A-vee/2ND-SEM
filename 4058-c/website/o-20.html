20. Create a class showing an example of default constructor. 


        #include <iostream>
        #include <string>
        
        class Example {
        private:
            int a;
            std::string b;
        
        public:
            Example() : a(0), b("default") {}
        
            void display() {
                std::cout << "a: " << a << ", b: " << b << std::endl;
            }
        };
        
        int main() {
            Example obj;
            obj.display();
        
            return 0;
        }

Output

a: 0, b: default
