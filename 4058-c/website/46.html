46. C++ Program To Calculate Batting Average


        #include<iostream>
        using namespace std;
        
        int main() {
            int match,run=0,r;
            cout<<"Enter number of matches player played : "<<endl;
            cin>>match;
            for(int i=0;i<match;i++) {
                cout<<"Enter run of match "<<i+1<<" : "<<endl;
                cin>>r;
                run=run+r;
            }
            float ba=(float)run/match;
            cout<<endl<<"Batting Average : "<<ba;
        }
