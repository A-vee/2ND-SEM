56. Write a program in C++ to display the first 5 natural numbers 


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n;
            cout << "Enter any number : ";
            cin >> n;
            cout << endl << "First " << n << " natural numbers are : ";
            for (int i = 1; i <= n; i++)
            {
                cout << endl << i;
            }
        }

Output


        Enter any number : 5
        
        First 5 natural numbers are : 
        1
        2
        3
        4
        5
