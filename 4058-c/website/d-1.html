1. C++ Program to convert String to int


        #include <iostream>
        #include <string>
        
        int main(){
            std::string st;
            std::cout << "Enter the integers for string : ";
            std::cin >> st;
            std::cout << "String value in integer : " << std::stoi(st);
            return 0;
        }

Output


            Enter the integers for string : 123
            String value in integer : 123
