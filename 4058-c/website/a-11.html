11. Program to print the number of elements present in an array


        #include<iostream>
        using namespace std;
        
        int main() {
            int arr[20],n,i,c=0;
            cout<<"Enter number of elements (1-20) : "<<endl;
            cin>>n;
            for(i=0;i<n;i++) {
                cout<<"Enter element "<<i+1<<" : ";
                cin>>arr[i];
                c++;
            }
            cout<<endl<<"Number of elements in array : "<<c;
        }

Output
