61. Write a C++ program that reads an integer and check whether it is negative, zero, or positive 


        #include<iostream>
        using namespace std;
        
        int main()
        {
            int n;
            cout << "Enter any number : ";
            cin >> n;
            if (n > 0)
            {
                cout << "Entered number is positive";
            }
            else if (n < 0)
            {
                cout << "Entered number is negative";
            }
            else
            {
                cout << "Entered number is zero";
            }
        }

Output


        Enter any number : 0
        Entered number is zero
